3. Write a “person” class to hold all the details.

Code:

class Person {
  constructor(FirstName, LastName) {
    this.firstName = FirstName;
    this.lastName = LastName;
  }

  sex(Gender) {
    this.gender = Gender;
  }

  age(Num) {
    this.num = Num;
  }

  contactNumber(ConNum) {
    this.conNum = ConNum;
  }

  mailID(Mail) {
    this.mail = Mail;
  }

  education(Qualification) {
    this.qualification = Qualification;
  }

  profession(Pro) {
    this.pro = Pro;
  }

  showDetails() {
    console.log(`
Name: ${this.firstName} ${this.lastName},
Sex: ${this.gender},
Age: ${this.num},
Contact Number: ${this.conNum},
Mail ID: ${this.mail},
Educational Qualification: ${this.qualification},
Profession: ${this.pro}

`);
  }
}

4. Write a class to calculate Uber price.

Code:

class Uber {
  uberAuto(km) {
    //Base Fare = 40/-
    //Charge Per Minute = 1/-
    //Charge Per km = 14.8/-
    let autoBaseFare = 40;
    let autoRideTimeCharge = 1;
    let autoCalc = autoBaseFare + autoRideTimeCharge * 1 + km * 14.8;
    console.log(
      `You have choosen AUTO and approximate cost of your ride is ${Math.trunc(
        autoCalc
      )}/-`
    );
  }

  uberCab(km) {
    //Base Fare = 45/-
    //Charge Per Minute = 1.5/-
    //Charge Per km = 20/-
    let cabBaseFare = 45;
    let cabRideTimeCharge = 1.5;
    let cabCalc = cabBaseFare + cabRideTimeCharge * 1.5 + km * 20;
    console.log(
      `You have choosen CAB and approximate cost of your ride is ${Math.trunc(
        cabCalc
      )}/-`
    );
  }

  uberPremium(km) {
    //Base Fare = 50/-
    //Charge Per Minute = 2/-
    //Charge Per km = 25/-
    let premiumBaseFare = 50;
    let premiumRideTimeCharge = 2;
    let premiumCalc = premiumBaseFare + premiumRideTimeCharge * 2 + km * 25;
    console.log(
      `You have choosen PREMIUM CAB and approximate cost of your ride is ${Math.trunc(
        premiumCalc
      )}/-`
    );
  }

  uberXL(km) {
    //Base Fare = 60/-
    //Charge Per Minute = 3/-
    //Charge Per km = 35/-
    let XLBaseFare = 60;
    let XLRideTimeCharge = 3;
    let XLCalc = XLBaseFare + XLRideTimeCharge * 3 + km * 35;
    console.log(
      `You have choosen SUV CAB and approximate cost of your ride is ${Math.trunc(
        XLCalc
      )}/-`
    );
  }
}

let auto = new Uber();
